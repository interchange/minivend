Configuration Examples


These files show configuration options for Apache similar to those used by the installer.

startup.pl 

use lib "/path/to/your/milton/lib";

=pod 
# This block must remain commented out if mod_perl is compiled as a
# DSO (loadable) module.
use DBI ();
use CGI;
use BASKET ();
use CONTAINER ();
use ITEM (); 
use MATRIX ();
use MILTON ();
use ORDER ();
use PAGE ();
use REGEN ();
use SEARCH ();
use SNIPPET ();
use SHIPPING ();
use AFFILIATES ();
use ORDERSTATS ();
=cut

1;

apache_include.conf 

#
# For Tallyman
#

# alias for where to find pages
Alias /tallyman/	"/usr/local/tallyman/tallyman/"

Alias /admin/		 "/usr/local/tallyman/admin/"
Alias /admin/images/ 	"/usr/local/tallyman/admin/images/"

PerlSetEnv TALLYMAN_PATH /usr/local/tallyman
PerlSetEnv TALLYMAN_SID uinta


    AllowOverride AuthConfig


PerlSetEnv EMBPERL_ESCMODE 0
SetEnv EMBPERL_OPTIONS 16
PerlSetEnv SESSION_FILE_DIRECTORY /tmp/sessions
PerlSetEnv EMBPERL_SESSION_CLASS  File
PerlModule Apache::Session::File
PerlModule HTML::Embperl
#PerlSetEnv EMBPERL_DEBUG 2285


	SetHandler  perl-script
	PerlHandler HTML::Embperl
	Options     ExecCGI


AddType text/html .epl
AddType text/html epl

PerlScript startup.pl
